# Karaoke Fighter — FULL BUILD PACK (Web + YouTube + 2 iPhones as 2 Mics + Lyrics + Street-Fighter HP)

> You asked for “the whole thing” so Claude can implement. This is a complete, end-to-end build plan + file tree + core code skeletons (server + host + mic + lyrics + pitch stability + Phaser fight + HP bars + quarter logic).

---

## What you’re building (MVP)
- **Laptop** opens `/host`:
  - Pastes **YouTube karaoke link**
  - Loads **LRC timed lyrics** (paste/upload OR fetch later)
  - Shows **Phaser fight scene** with **HP bars**
  - Splits song into **4 equal quarters** based on YouTube duration
  - Displays current lyric line at bottom
  - Shows who is **ATTACK** (singer) vs **DEFEND**
  - Receives `p1Score` and `p2Score` from two phones
  - Applies damage every tick (0.5s) using attacker score

- **iPhone #1** opens `/mic?p=1` and **iPhone #2** opens `/mic?p=2`:
  - Asks mic permission
  - Computes a **pitch stability score** (0–1)
  - Streams score to host via WebSocket

### Hard Rules
- iPhone mic requires **HTTPS** (use ngrok).
- **Lyrics must exist**: game cannot start unless LRC loaded and parsed.
- YouTube provides playback + timestamps; do NOT try to rip audio.

import express from "express";
import http from "http";
import path from "path";
import { fileURLToPath } from "url";
import { Server } from "socket.io";

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

const app = express();
const server = http.createServer(app);
const io = new Server(server);

// Serve static client files
app.use(express.static(path.join(__dirname, "../client")));

// Routes
app.get("/host", (_, res) => res.sendFile(path.join(__dirname, "../client/host.html")));
app.get("/mic", (_, res) => res.sendFile(path.join(__dirname, "../client/mic.html")));

// Socket routing
let hostSocketId = null;

io.on("connection", (socket) => {
  socket.on("registerHost", () => {
    hostSocketId = socket.id;
    socket.emit("hostRegistered", { ok: true });
  });

  socket.on("registerMic", ({ player }) => {
    socket.data.player = player;
    socket.emit("micRegistered", { ok: true, player });
    if (hostSocketId) io.to(hostSocketId).emit("micStatus", { player, connected: true });
  });

  socket.on("score", ({ player, score }) => {
    if (hostSocketId) io.to(hostSocketId).emit("score", { player, score });
  });

  socket.on("disconnect", () => {
    if (socket.id === hostSocketId) hostSocketId = null;
    const p = socket.data.player;
    if (p && hostSocketId) io.to(hostSocketId).emit("micStatus", { player: p, connected: false });
  });
});

const PORT = process.env.PORT || 5173;
server.listen(PORT, () => console.log(`Server running on http://localhost:${PORT}`));

4) Shared Utilities (YouTube ID + LRC Parser)

client/shared.js

export function extractYouTubeId(url) {
  // supports youtu.be/<id> or youtube.com/watch?v=<id> or embed/<id>
  try {
    const u = new URL(url.trim());
    if (u.hostname.includes("youtu.be")) return u.pathname.slice(1);
    if (u.searchParams.get("v")) return u.searchParams.get("v");
    const parts = u.pathname.split("/").filter(Boolean);
    const embedIndex = parts.indexOf("embed");
    if (embedIndex >= 0 && parts[embedIndex+1]) return parts[embedIndex+1];
  } catch {}
  return null;
}

export function parseLRC(lrcText) {
  // Returns sorted array: [{t: secondsFloat, text}]
  // LRC lines: [mm:ss.xx] lyric text
  const lines = [];
  const raw = lrcText.split(/\r?\n/);
  const timeRe = /$begin:math:display$\(\\d\{1\,2\}\)\:\(\\d\{2\}\)\(\?\:\\\.\(\\d\{1\,2\}\)\)\?$end:math:display$/g;

  for (const line of raw) {
    let match;
    let text = line.replace(timeRe, "").trim();
    while ((match = timeRe.exec(line)) !== null) {
      const mm = parseInt(match[1], 10);
      const ss = parseInt(match[2], 10);
      const xx = match[3] ? parseInt(match[3], 10) : 0;
      const t = mm * 60 + ss + (xx / (match[3]?.length === 1 ? 10 : 100));
      lines.push({ t, text });
    }
  }
  lines.sort((a,b) => a.t - b.t);
  return lines.filter(x => Number.isFinite(x.t));
}

export function currentLyricAtTime(lines, t) {
  if (!lines?.length) return "";
  // simple linear scan from end could be optimized (binary search)
  let best = "";
  for (let i = 0; i < lines.length; i++) {
    if (lines[i].t <= t) best = lines[i].text;
    else break;
  }
  return best;
}

5) Phaser Scene (fighters + HP bars + animations)

client/phaserScene.js

export function createPhaserGame({ mountId, onReady }) {
  let sceneRef;

  const config = {
    type: Phaser.AUTO,
    width: 800,
    height: 450,
    parent: mountId,
    backgroundColor: "#000000",
    scene: {
      preload,
      create,
      update
    }
  };

  const game = new Phaser.Game(config);

  function preload() {
    this.load.image("bg", "/assets/bg.png");
    // You MUST provide sprite sheets with frame sizes matching your assets
    this.load.spritesheet("p1", "/assets/p1.png", { frameWidth: 80, frameHeight: 80 });
    this.load.spritesheet("p2", "/assets/p2.png", { frameWidth: 80, frameHeight: 80 });
  }

  function create() {
    sceneRef = this;

    this.add.image(400, 225, "bg").setScale(1.0);

    // Fighters
    this.p1 = this.add.sprite(220, 320, "p1").setScale(2);
    this.p2 = this.add.sprite(580, 320, "p2").setScale(2).setFlipX(true);

    // Animations (adjust frame ranges to match your sprite sheet)
    this.anims.create({ key: "p1-idle", frames: this.anims.generateFrameNumbers("p1", { start: 0, end: 3 }), frameRate: 8, repeat: -1 });
    this.anims.create({ key: "p1-punch", frames: this.anims.generateFrameNumbers("p1", { start: 4, end: 7 }), frameRate: 12 });
    this.anims.create({ key: "p1-hit", frames: this.anims.generateFrameNumbers("p1", { start: 8, end: 11 }), frameRate: 12 });
    this.anims.create({ key: "p1-block", frames: this.anims.generateFrameNumbers("p1", { start: 12, end: 15 }), frameRate: 12 });

    this.anims.create({ key: "p2-idle", frames: this.anims.generateFrameNumbers("p2", { start: 0, end: 3 }), frameRate: 8, repeat: -1 });
    this.anims.create({ key: "p2-punch", frames: this.anims.generateFrameNumbers("p2", { start: 4, end: 7 }), frameRate: 12 });
    this.anims.create({ key: "p2-hit", frames: this.anims.generateFrameNumbers("p2", { start: 8, end: 11 }), frameRate: 12 });
    this.anims.create({ key: "p2-block", frames: this.anims.generateFrameNumbers("p2", { start: 12, end: 15 }), frameRate: 12 });

    this.p1.play("p1-idle");
    this.p2.play("p2-idle");

    // HP
    this.p1HP = 100;
    this.p2HP = 100;

    const BAR_W = 300, BAR_H = 18;

    // Background bars
    this.add.rectangle(40, 30, BAR_W, BAR_H, 0x222222).setOrigin(0, 0.5);
    this.add.rectangle(760, 30, BAR_W, BAR_H, 0x222222).setOrigin(1, 0.5);

    // Fill bars
    this.p1Bar = this.add.rectangle(40, 30, BAR_W, BAR_H, 0xff0000).setOrigin(0, 0.5);
    this.p2Bar = this.add.rectangle(760, 30, BAR_W, BAR_H, 0xff0000).setOrigin(1, 0.5);

    this.add.text(40, 6, "P1", { fontSize: "14px", color: "#fff" });
    this.add.text(760, 6, "P2", { fontSize: "14px", color: "#fff" }).setOrigin(1, 0);

    this.koText = this.add.text(400, 180, "", { fontSize: "72px", color: "#fff" }).setOrigin(0.5);
    this.gameOver = false;

    // Attack API exposed to host.js
    this.doAttack = ({ attacker, score }) => {
      if (this.gameOver) return;

      const isP1 = attacker === 1;
      const atk = isP1 ? this.p1 : this.p2;
      const def = isP1 ? this.p2 : this.p1;

      const base = 3, bonus = 12;
      let damage = base + bonus * score;

      let blocked = false;
      if (score < 0.35) {
        blocked = true;
        damage *= 0.4;
      }

      if (isP1) this.p2HP = Math.max(0, this.p2HP - damage);
      else this.p1HP = Math.max(0, this.p1HP - damage);

      this.updateBars();

      // Animations
      atk.play(isP1 ? "p1-punch" : "p2-punch");
      def.play(blocked ? (isP1 ? "p2-block" : "p1-block") : (isP1 ? "p2-hit" : "p1-hit"));

      // “Street Fighter feel”
      const strong = score > 0.8;
      this.cameras.main.shake(80, strong ? 0.01 : 0.004);

      atk.once("animationcomplete", () => atk.play(isP1 ? "p1-idle" : "p2-idle"));
      def.once("animationcomplete", () => def.play(isP1 ? "p2-idle" : "p1-idle"));

      // KO
      if (this.p1HP <= 0 || this.p2HP <= 0) {
        this.gameOver = true;
        this.koText.setText("KO");
      }
    };

    this.updateBars = () => {
      this.p1Bar.width = 300 * (this.p1HP / 100);
      this.p2Bar.width = 300 * (this.p2HP / 100);
    };

    if (onReady) onReady(this);
  }

  function update() {}

  return {
    getScene: () => sceneRef
  };
}
